116 inline uint64_t CheckedFile::logicalToPhysical( uint64_t logicalOffset )
117 {
118 const uint64_t page = logicalOffset / logicalPageSize;
119 const uint64_t remainder = logicalOffset - page * logicalPageSize;
120
121 return page * physicalPageSize + remainder;
122 }
123
124 inline uint64_t CheckedFile::physicalToLogical( uint64_t physicalOffset )
125 {
126 const uint64_t page = physicalOffset >> physicalPageSizeLog2;
127 const size_t remainder = static_cast<size_t>( physicalOffset & physicalPageSizeMask );
128
129 return page * logicalPageSize + std::min( remainder, logicalPageSize );
130 }

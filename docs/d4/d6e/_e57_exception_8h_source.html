305 public:
306 const char *what() const noexcept override
307 {
308 return "E57 exception";
309 }

311 void report( const char *reportingFileName = nullptr, int reportingLineNumber = 0,
312 const char *reportingFunctionName = nullptr,
313 std::ostream &os = std::cout ) const noexcept
314 {
315 os << "**** Got an e57 exception: " << errorStr() << std::endl;
316
317#ifdef E57_ENABLE_DIAGNOSTIC_OUTPUT
318 os << " Debug info: " << std::endl;
319 os << " context: " << context_ << std::endl;
320 os << " sourceFunctionName: " << sourceFunctionName_ << std::endl;
321 if ( reportingFunctionName != nullptr )
322 {
323 os << " reportingFunctionName: " << reportingFunctionName << std::endl;
324 }
325
326 /*** Add a line in error message that a smart editor (gnu emacs) can
327 * interpret as a link to the source code: */
328 os << sourceFileName_ << "(" << sourceLineNumber_ << ") : error C" << errorCode_
329 << ": <--- occurred on" << std::endl;
330 if ( reportingFileName != nullptr )
331 {
332 os << reportingFileName << "(" << reportingLineNumber
333 << ") : error C0: <--- reported on" << std::endl;
334 }
335#else
336 UNUSED( reportingFileName );
337 UNUSED( reportingLineNumber );
338 UNUSED( reportingFunctionName );
339#endif
340 }
